#### [76. 最小覆盖子串](https://leetcode-cn.com/problems/minimum-window-substring/)

难度困难775

给你一个字符串 S、一个字符串 T 。请你设计一种算法，可以在 O(n) 的时间复杂度内，从字符串 S 里面找出：包含 T 所有字符的最小子串。

 

**示例：**

```
输入：S = "ADOBECODEBANC", T = "ABC"
输出："BANC"
```

 

**提示：**

- 如果 S 中不存这样的子串，则返回空字符串 `""`。
- 如果 S 中存在这样的子串，我们保证它是唯一的答案。







## 超时

```c++
class Solution {
public:
    string minWindow(string s, string t) {
        int t_flag[128], cur_flag[128];
        memset(t_flag, 0, sizeof(int) * 128);

        int l = 0, r = 0, min_size = INT_MAX, remain = t.size(), result_l;
        string result = "";

        for(int i = 0; i < t.size(); ++i)
            ++t_flag[t[i]];
        
        while(l < s.size() - t.size() + 1) {
            if(t_flag[s[l]] > 0) {
                // reset
                
                remain = t.size();
                memcpy(cur_flag, t_flag, sizeof(int) * 128);

                cur_flag[s[l]]--;
                remain--;

                if (remain == 0 && min_size > r - l + 1) {
                    min_size = 1;
                    result_l = l;
                    return s.substr(result_l, min_size);
                }

                for(auto r = l + 1; r < s.size(); ++r) {
                    if (cur_flag[s[r]] > 0) {
                        cur_flag[s[r]]--;
                        remain--;
                        if (remain == 0 && min_size > r - l + 1) {
                            min_size = r - l + 1;
                            result_l = l;
                            if (min_size == t.size())
                                return s.substr(result_l, min_size);
                        }
                    }
                }
            } 
            ++l;
        }
        return min_size == INT_MAX ? "" : s.substr(result_l, min_size);
    }
};
```

class Solution {
public:
    string minWindow(string s, string t) {
        int t_flag[128], cur_flag[128];
        int s_size = s.size(), t_size = t.size();
        memset(t_flag, 0, sizeof(int) * 128);
        memset(cur_flag, 0, sizeof(int) * 128);

        int l = 0, r = 0, min_size = INT_MAX, remain = t_size, result_l;

        for(int i = 0; i < t.size(); ++i) {
            ++t_flag[t[i]];
            ++cur_flag[t[i]];
        }
        
        while(r < s_size) {
            if (t_flag[s[r]] > 0) {
                if (--cur_flag[s[r]] >= 0)
                    remain--;
            }
            ++r;
            while(remain == 0) {
                cout << s[l] << endl;
                if(r - l == t_size)
                    return s.substr(l, t_size);
                if(r - l < min_size) {
                    min_size = r - l;
                    result_l = l;
                }
                if(t_flag[s[l]] > 0) {
                    if(++cur_flag[s[l]] > 0) 
                        remain++;
                }
                ++l;
            } 
        }
        
        return min_size == INT_MAX ? "" : s.substr(result_l, min_size);
    }
};
